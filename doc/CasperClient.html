def set_ipaddress_and_port_number
  puts "Enter IP Address:"
  @ip_address = gets.chomp    
  
  puts "Enter Port Number:"   
  @port = gets.chomp.to_i
  
  self.url = "http://" + self.ip_address + ":" + self.port.to_s + "/rpc"
end
